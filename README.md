### sonata-ui-core



### Install

```
npm install sonata-ui-core
```

### Usage

```
import React from 'react';
import TButton  from 'sonata-ui-core';

const  App=()=> {
  return (
    <div className="App">
           <h1>Button from @sonata-ui-core package</h1>
           <TButton >Primary Button</TButton>
    </div>
  );
}

export default App;
```
